64bool SerialCommandExecuter::_parseCommand(String commandString) {
65 String* trimmedCmdLine = _trimCommand(commandString);
66 String command = trimmedCmdLine[0].c_str();
67 String params[MAX_NUMBER_PARAMS] = {""};
68 uint8_t numParams = 0;
69
70 while (numParams < MAX_NUMBER_PARAMS) {
71 if (trimmedCmdLine[numParams+1] == "") { //+1, because the command is in the first cell
72 break;
73 }
74 numParams++;
75 }
76
77 for (uint8_t i = 1; i-1 < numParams; i++){ //+1, because the command is in the first cell
78 params[i-1] = trimmedCmdLine[i].c_str();
79 }
80
81 /* Check which command is given */
82 if (command == COMMAND_1) {
83 _printHelp();
84 } else if (command == COMMAND_2) {
85 if (!_checkParams(numParams, 1) || !_enableDebug(params[0])) {
86 return false;
87 }
88 } else if (command == COMMAND_3) {
89 if (!_checkParams(numParams, 1) || !_superUserLogin(params[0])) {
90 return false;
91 }
92 } else if (command == COMMAND_4) {
93 if (!_viewKey()) {
94 return false;
95 }
96 } else if (command == COMMAND_5) {
97 _restart();
98 } else if (command == COMMAND_6) {
99 if (!_viewUsers()) {
100 return false;
101 }
102 } else {
103 Serial.println(ERROR_2_TEXT);
104 return false;
105 }
106 return true;
107}

116String* SerialCommandExecuter::_trimCommand(String commandString) {
117 static String commandItems[1+MAX_NUMBER_PARAMS] = {""}; //To save command and parameters, each in own cell
118 String item = ""; //Can be a command or parameter
119 uint8_t numParams = 1; //1, because the command counts as well
120 uint8_t paramCounter = 0;
121 uint16_t c = 0;
122
123 /* Reset static array */
124 for (uint16_t x = 0; x < 1+MAX_NUMBER_PARAMS; x++) {
125 commandItems[x] = "";
126 }
127
128 /* Count number of parameters */
129 for (uint16_t c = 0; c < commandString.length(); c++) {
130 if (commandString[c] == '-') {
131 numParams++;
132 }
133 }
134
135 if (numParams-1 > MAX_NUMBER_PARAMS) {
136 numParams == MAX_NUMBER_PARAMS;
137 }
138
139 while (paramCounter < numParams) {
140 /* If c == 0, the command is extracted */
141 if (c != 0) {
142 if (commandString.indexOf('-', c) != -1) {
143 c = commandString.indexOf('-', c) + 1;
144 } else {
145 break;
146 }
147 }
148 while (commandString[c] != ' ' && commandString[c] != '-' && commandString[c] != '\n') {
149 item += commandString[c];
150 c++;
151 }
152 commandItems[paramCounter] = item; //Save param to items list
153 item = ""; //Reset param
154 paramCounter++;
155 }
156 return commandItems;
157}
