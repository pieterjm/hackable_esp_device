62bool SerialCommandExecuter::_parseCommand(String commandString) {
63 String* trimmedCmdLine = _trimCommand(commandString);
64 String command = trimmedCmdLine[0].c_str();
65 String params[MAX_NUMBER_PARAMS] = {""};
66 uint8_t numParams = 0;
67
68 while (numParams < MAX_NUMBER_PARAMS) {
69 if (trimmedCmdLine[numParams+1] == "") { //+1, because the command is in the first cell
70 break;
71 }
72 numParams++;
73 }
74
75 for (uint8_t i = 1; i-1 < numParams; i++){ //+1, because the command is in the first cell
76 params[i-1] = trimmedCmdLine[i].c_str();
77 }
78
79 /* Check which command is given */
80 if (command == COMMAND_HELP) {
81 _printHelp(COMMAND_HELP);
82 } else {
83 /* If help needs to be printed, print it and return */
84 if (_checkHelp(params[0], command)) {
85 return true;
86 }
87 }
88
89 if (command == COMMAND_DEBUG) {
90 if (!_checkParams(numParams, 1, 1) || !_enableDebug(params[0])) {
91 return false;
92 }
93 } else if (command == COMMAND_SU) {
94 if (!_checkParams(numParams, 1, 1) || !_superUserLogin(params[0])) {
95 return false;
96 }
97 } else if (command == COMMAND_KEYS) {
98 if (!_viewKey()) {
99 return false;
100 }
101 } else if ((command == COMMAND_RESTART)) {
102 _restart();
103 } else if (command == COMMAND_USERS) {
104 if (!_viewUsers()) {
105 return false;
106 }
107 } else if (command == COMMAND_HOSTNAME) {
108 if (!_checkParams(numParams, 0, 2) || !_hostname(params)) {
109 return false;
110 }
111 } else if (command == COMMAND_LS) {
112 buffOverflow.ls();
113 } else if (command == COMMAND_VI) {
114 if (_checkParams(numParams, 1, 1)) {
115 if (params[0] == "./testprogram.c" || params[0] == "testprogram.c") {
116 buffOverflow.vi();
117 } else {
118 Serial.println(ERROR_NO_FILE);
119 return false;
120 }
121 }
122 } else if (command.substring(0, 2) == COMMAND_RUN) { //Substring == "./" the rest is filename
123 if (_checkParams(numParams, 0, 1)) {
124 String filename = command.substring(2); //The rest of the command is filename
125
126 if (filename == "testprogram.c") {
127 Serial.println(ERROR_PERM_DENIED);
128 return false;
129 }
130

138 * user is logged in.
139 */

141 _isLoggedIn = true;
142 Serial.println(MESS_SUPER_USER);
143 }
144 } else {
145 buffOverflow.runCProgram("");
146 }
147 }
148 } else if (command == COMMAND_OBJDUMP) {
149 if (_checkParams(numParams, 2, 2)) {
150 if (params[0] != "-d") {
151 Serial.println(ERROR_WRONG_ARGS);
152 return false;
153 }
154 if (params[1] == "./testprogram" || params[1] == "testprogram") {
155 buffOverflow.objectDump();
156 } else {
157 Serial.println(ERROR_NO_FILE);
158 return false;
159 }
160 }
161 } else {
162 Serial.println(ERROR_CMD_NOT_FOUND);
163 return false;
164 }

175String* SerialCommandExecuter::_trimCommand(String commandString) {
176 static String commandItems[1+MAX_NUMBER_PARAMS] = {""}; //To save command and parameters, each in own cell
177 String item = ""; //Can be a command or parameter
178 uint8_t paramCounter = 0;
179
180 /* Reset static array */
181 for (uint16_t x = 0; x < 1+MAX_NUMBER_PARAMS; x++) {
182 commandItems[x] = "";
183 }
184
185 /* Count number of parameters by adding to temp variable if not a whitespace or end of line*/
186 for (uint16_t c = 0; c < commandString.length(); c++) {
187 if (commandString[c] == ' ' || commandString[c] == '\n') {
188 /* If item is not empty: add to item array */
189 if (item != ""){
190 commandItems[paramCounter] = item; //Save param to items list
191 item = ""; //Reset item value
192 paramCounter++;
193 }
194 } else { // if not a whitepace add to item
195 item += commandString[c];
196 }
197 }
198 return commandItems;
199}
