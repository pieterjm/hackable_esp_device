126static void xor_block (byte * d, byte * s)
127{
128 for (byte i = 0 ; i < N_BLOCK ; i += 4)
129 {
130 *d++ ^= *s++ ; // some unrolling
131 *d++ ^= *s++ ;
132 *d++ ^= *s++ ;
133 *d++ ^= *s++ ;
134 }
135}

137static void copy_and_key (byte * d, byte * s, byte * k)
138{
139 for (byte i = 0 ; i < N_BLOCK ; i += 4)
140 {
141 *d++ = *s++ ^ *k++ ; // some unrolling
142 *d++ = *s++ ^ *k++ ;
143 *d++ = *s++ ^ *k++ ;
144 *d++ = *s++ ^ *k++ ;
145 }
146}

152static void shift_sub_rows (byte st [N_BLOCK])
153{
154 st [0] = s_box (st [0]) ; st [4] = s_box (st [4]) ;
155 st [8] = s_box (st [8]) ; st [12] = s_box (st [12]) ;
156
157 byte tt = st [1] ;
158 st [1] = s_box (st [5]) ; st [5] = s_box (st [9]) ;
159 st [9] = s_box (st [13]) ; st [13] = s_box (tt) ;
160
161 tt = st[2] ; st [2] = s_box (st [10]) ; st [10] = s_box (tt) ;
162 tt = st[6] ; st [6] = s_box (st [14]) ; st [14] = s_box (tt) ;
163
164 tt = st[15] ;

166 st [7] = s_box (st [3]) ; st [3] = s_box (tt) ;

186/* SUB COLUMNS PHASE */
187
188static void mix_sub_columns (byte dt[N_BLOCK], byte st[N_BLOCK])
189{
190 byte j = 5 ;
191 byte k = 10 ;
192 byte l = 15 ;
193 for (byte i = 0 ; i < N_BLOCK ; i += N_COL)
194 {
195 byte a = st [i] ;
196 byte b = st [j] ; j = (j+N_COL) & 15 ;
197 byte c = st [k] ; k = (k+N_COL) & 15 ;
198 byte d = st [l] ; l = (l+N_COL) & 15 ;
199 byte a1 = s_box (a), b1 = s_box (b), c1 = s_box (c), d1 = s_box (d) ;
200 byte a2 = f2(a1), b2 = f2(b1), c2 = f2(c1), d2 = f2(d1) ;
201 dt[i] = a2 ^ b2^b1 ^ c1 ^ d1 ;
202 dt[i+1] = a1 ^ b2 ^ c2^c1 ^ d1 ;
203 dt[i+2] = a1 ^ b1 ^ c2 ^ d2^d1 ;
204 dt[i+3] = a2^a1 ^ b1 ^ c1 ^ d2 ;
205 }
206}
207
208static void inv_mix_sub_columns (byte dt[N_BLOCK], byte st[N_BLOCK])
209{
210 for (byte i = 0 ; i < N_BLOCK ; i += N_COL)
211 {
212 byte a1 = st [i] ;
213 byte b1 = st [i+1] ;
214 byte c1 = st [i+2] ;
215 byte d1 = st [i+3] ;
216 byte a2 = f2(a1), b2 = f2(b1), c2 = f2(c1), d2 = f2(d1) ;
217 byte a4 = f2(a2), b4 = f2(b2), c4 = f2(c2), d4 = f2(d2) ;
218 byte a8 = f2(a4), b8 = f2(b4), c8 = f2(c4), d8 = f2(d4) ;
219 byte a9 = a8 ^ a1,b9 = b8 ^ b1,c9 = c8 ^ c1,d9 = d8 ^ d1 ;
220 byte ac = a8 ^ a4,bc = b8 ^ b4,cc = c8 ^ c4,dc = d8 ^ d4 ;
221
222 dt[i] = is_box (ac^a2 ^ b9^b2 ^ cc^c1 ^ d9) ;
223 dt[(i+5)&15] = is_box (a9 ^ bc^b2 ^ c9^c2 ^ dc^d1) ;
224 dt[(i+10)&15] = is_box (ac^a1 ^ b9 ^ cc^c2 ^ d9^d2) ;
225 dt[(i+15)&15] = is_box (a9^a2 ^ bc^b1 ^ c9 ^ dc^d2) ;
226 }
227}

254
255byte AES::set_key (byte key [], int keylen)
256{
257 byte hi ;
258 switch (keylen)
259 {
260 case 16:
261 case 128:
262 keylen = 16; // 10 rounds
263 round = 10 ;
264 break;
265 case 24:
266 case 192:
267 keylen = 24; // 12 rounds
268 round = 12 ;
269 break;
270 case 32:
271 case 256:
272 keylen = 32; // 14 rounds
273 round = 14 ;
274 break;
275 default:
276 round = 0;
277 return FAILURE;
278 }
279 hi = (round + 1) << 4 ;
280 copy_n_bytes (key_sched, key, keylen) ;
281 byte t[4] ;
282 byte next = keylen ;
283 for (byte cc = keylen, rc = 1 ; cc < hi ; cc += N_COL)
284 {
285 for (byte i = 0 ; i < N_COL ; i++)
286 t[i] = key_sched [cc-4+i] ;
287 if (cc == next)
288 {
289 next += keylen ;
290 byte ttt = t[0] ;
291 t[0] = s_box (t[1]) ^ rc ;
292 t[1] = s_box (t[2]) ;
293 t[2] = s_box (t[3]) ;
294 t[3] = s_box (ttt) ;
295 rc = f2 (rc) ;
296 }
297 else if (keylen == 32 && (cc & 31) == 16)
298 {
299 for (byte i = 0 ; i < 4 ; i++)
300 t[i] = s_box (t[i]) ;
301 }
302 byte tt = cc - keylen ;
303 for (byte i = 0 ; i < N_COL ; i++)
304 key_sched [cc + i] = key_sched [tt + i] ^ t[i] ;
305 }
306 return SUCCESS ;
307}

320void AES::copy_n_bytes (byte * d, byte * s, byte nn)
321{
322 while (nn >= 4)
323 {
324 *d++ = *s++ ; // some unrolling
325 *d++ = *s++ ;
326 *d++ = *s++ ;
327 *d++ = *s++ ;
328 nn -= 4 ;
329 }
330 while (nn--)
331 *d++ = *s++ ;
332}

336byte AES::encrypt (byte plain [N_BLOCK], byte cipher [N_BLOCK])
337{
338 if (round)
339 {
340 byte s1 [N_BLOCK], r ;
341 copy_and_key (s1, plain, (byte*) (key_sched)) ;
342
343 for (r = 1 ; r < round ; r++)
344 {
345 byte s2 [N_BLOCK] ;
346 mix_sub_columns (s2, s1) ;
347 copy_and_key (s1, s2, (byte*) (key_sched + r * N_BLOCK)) ;
348 }
349 shift_sub_rows (s1) ;
350 copy_and_key (cipher, s1, (byte*) (key_sched + r * N_BLOCK)) ;
351 }
352 else
353 return FAILURE ;
354 return SUCCESS ;
355}

392{

394 {
395 byte s1 [N_BLOCK] ;
396 copy_and_key (s1, plain, (byte*) (key_sched + round * N_BLOCK)) ;
397 inv_shift_sub_rows (s1) ;
398
399 for (byte r = round ; --r ; )

401 byte s2 [N_BLOCK] ;
402 copy_and_key (s2, s1, (byte*) (key_sched + r * N_BLOCK)) ;
403 inv_mix_sub_columns (s1, s2) ;
404 }
405 copy_and_key (cipher, s1, (byte*) (key_sched)) ;
406 }
407 else
408 return FAILURE ;
409 return SUCCESS ;
