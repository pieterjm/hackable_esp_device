163

169 /* Load style_mobile.css file, styling for mobile version */

175 server.on("/style_switch.css", HTTP_GET, []() {

181 handleFileRequest("/favicon.ico", PERMISSION_LVL_ALL);

183
184 /* Load jquery.min.js file, for ajax */
185 server.on("/jquery.min.js", HTTP_GET, []() {
186 handleFileRequest("/jquery.min.js", PERMISSION_LVL_ALL);
187 });
188
189 /* Load base.js file, JavaScript for site */
190 server.on("/base.js", HTTP_GET, []() {
191 handleFileRequest("/base.js", PERMISSION_LVL_ALL);
192 });
193
194 /* Load switch.js file, JavaScript for on/off switch */
195 server.on("/switch.js", HTTP_GET, []() {
196 handleFileRequest("/switch.js", PERMISSION_LVL_ALL);
197 });
198 /*
199 * End of file loading
200 */
201
202 /*
203 * Routes for JavaScript data receiving
204 */
205 /* Route for setting power */
206 server.on("/set_power", HTTP_GET, []() {
207 if (server.arg("state")) {
208 ledState = atoi(server.arg("state").c_str());
209 digitalWrite(LED_BUILTIN, !ledState);
210 }
211 handleFileRequest("/index.html", PERMISSION_LVL_ALL);
212 });
213
214 /* Route for restarting the server */
215 server.on("/restart", HTTP_GET, []() {
216 handleFileRequest("/", PERMISSION_LVL_ALL);
217 ESP.restart();
218 });
219 /*
220 * End of JavaScript data receiving
221 */
222 /*
223 * Routes for file management
224 */
225 /* Route for file upload request */
226 server.on("/upload", HTTP_POST, []() {
227 server.send(200); //HTTP code 200 == OK
228 debugln("Wait, something got uploaded");
229 }, handleFileUpload //Receive and save the file
230 );
231 /* Route for file upload request */
232 server.on("/download", HTTP_POST, []() {
233 debugln("File download request");
234 }, handleFileDownload //Receive and save the file
235 );
236 /*
237 * End of routes for file management
238 */
239 /* Not found */
240 server.onNotFound([]() { //If the client requests any URI
241 handleFileRequest(server.uri(), PERMISSION_LVL_ALL); //send it if it exists
242 debugln("NOT_FOUND?");
243 });
244 server.begin(); //Start server
245}

290String getContentType(String filename) {
291 if (filename.endsWith(".html")) return "text/html";
292 else if (filename.endsWith(".css")) return "text/css";
293 else if (filename.endsWith(".js")) return "application/javascript";
294 else if (filename.endsWith(".ico")) return "image/x-icon";
295 else if (filename.endsWith(".gz")) return "application/x-gzip";
296 else if (filename.endsWith(".txt")) return "text/plain";
297 return "text/plain";
298}

307void handleFileRequest(String path, uint8_t permissionLevel) {
308 if(!userHandler.checkPermission(permissionLevel, &server)) {
309 server.requestAuthentication();
310 return;
311 }
312
313 debugln(String("Requested file: ") + path);
314
315 String contentType = getContentType(path); //Get the MIME type
316 String pathWithGz = path + ".gz";
317
318 if (SPIFFS.exists(pathWithGz)) { //If there's a compressed version available
319 path += ".gz"; //Use the compressed verion
320 }
321
322 if (SPIFFS.exists(path)) {
323 File file = SPIFFS.open(path, "r"); //Open the file
324 size_t sent = server.streamFile(file, contentType); //Send it to the client
325 file.close(); //Close the file again
326 debugln(String("Sent file: ") + path);
327 return;
328 }
329
330 debugln(String("File Not Found: ") + path); //If the file doesn't exist, return false
331 server.send(404, "text/plain", "404: Not Found"); //otherwise, respond with a 404 (Not Found) error
332}

338/**************************************************************************/
339void handleFileUpload() {
340 HTTPUpload& upload = server.upload();
341
342 if (upload.status == UPLOAD_FILE_START) {
343 String filename = upload.filename;
344
345 if (!filename.startsWith("/")) {
346 filename = "/" + filename;
347 }
348
349 debugln(String("Upload file named: ") + filename);
350
351 fsUploadFile = SPIFFS.open(filename, "w"); //Open the file for writing in SPIFFS (create if it doesn't exist)
352
353 } else if (upload.status == UPLOAD_FILE_WRITE && fsUploadFile ) {
354 fsUploadFile.write(upload.buf, upload.currentSize); //Write the received bytes to the file
355 } else if (upload.status == UPLOAD_FILE_END) {
356 if (fsUploadFile) { //If the file was successfully created
357 fsUploadFile.close(); //Close the file again
358 debugln(String("handleFileUpload Size: ") + upload.totalSize);
359 server.sendHeader("Location","/success.html"); //Redirect the client to the success page
360 server.send(303);
361 userHandler.updateUsers();
362 cliExecuter.setUsers(userHandler.getUsers(), userHandler.getNumberOfUsers()); //Update users for cli as well
363 } else {
364 server.send(500, "text/plain", "500: couldn't create file");
365 }
366 }
367}

374void handleFileDownload() {
375 String filename = server.arg("filekey"); //Get user input for filename
376
377 if(!filename.startsWith("/")) {
378 filename = "/" + filename;
379 }
380
381 if (!SPIFFS.exists(filename)) {
382 server.send(404, "text/plain", "404: file not found!");
383 return;
384 }
385
386 File download = SPIFFS.open(filename, "r");
387
388 debugln("Start sending file");
389
390 server.sendHeader("Content-Type", "text/text");
391 server.sendHeader("Content-Disposition", "attachment; filename="+filename);
392 server.sendHeader("Connection", "close");
393 server.streamFile(download, "application/octet-stream");
394 download.close();
395 server.send(200); //HTTP code 200 == OK
396}
